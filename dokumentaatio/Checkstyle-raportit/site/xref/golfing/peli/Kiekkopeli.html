15  /**
16   * Kiekkopeli on itse peli luokka. Se on yhteydessä käyttäjälle näkyvään
17   * ulkoasuun ja näppäimistönkuuntelijaan. Sillä on myös yhteys Ratoihin, jotta
18   * se tietää missä kunkin väylän kori sijaitsee. Se tuntee pelaajan ja pelaajan
19   * kautta heiton.
20   *
21   * Luokka on vastuussa kaikesta mitä pelissä tapahtuu ja käyttää muita luokkia
22   * apunaan tässä tehtävässä.
23   *
24   */
25  public class Kiekkopeli extends Timer implements ActionListener {
26  
27      private int leveys;
28      private int korkeus;
29      private boolean jatkuu;
30      private boolean liianlujaa;
31      private Paivitettava paivitettava;
32      private Rata rata;

34      private Kiekko kiekko;
35      private Kori kori;

37      private String tilanne;
38      private String viesti;

40      public Kiekkopeli(int leveys, int korkeus, Pelaaja pelaaja, String rata, Radat radat) {
41          super(1000, null);
42  
43          this.pelaaja = pelaaja;
44          this.leveys = leveys;
45          this.korkeus = korkeus;
46          this.jatkuu = true;
47          this.liianlujaa = false;
48          this.kiekko = this.pelaaja.getKiekko("draiveri");
49          this.heitto = pelaaja.getHeitto();
50  
51          this.rata = radat.getRadat().get(rata);
52          this.kori = this.rata.getVaylat().get(4).getKori();
53  
54          addActionListener(this);
55          setInitialDelay(1000);
56  
57      }

160     /**
161      * Palauttaa viestin.
162      * Alussa viesti on tervehdys pelaajalle.
163      * Jos kiekko osuu liian suurella voimalla koriin viestitetään tästä.
164      * Lopuksi viestitetään kiekon olevan korissa ja heittojenmäärä joka kului.
165      */
166     public String getViesti() {
167         if (this.pelaaja.montakoHeittoa() == 0) {
168             return "Tervehdys " + pelaaja.getNimi() + "! (:\nVaro puuta!";
169         } else if (!jatkuu && !liianlujaa) {
170             return "KORISSA! \nSelvitit väylän " + pelaaja.montakoHeittoa() + " heitolla.";
171         } else if (!jatkuu && liianlujaa) {
172             return "Kiekko tippui korista! \nJoudutaan heittämään vielä kerran...";
173         } else {
174             return " \n ";
175         }
176     }

178     /**
179      * Muuttaa pelin tilannetta.
180      * Käskee myös paivitettavaa päivittämään tilanne ruudulle.
181      * Jos heitto on lähetetty kulkee se niin kauan kuin sillä on voimaa, tai 
182      * kunnes se osuu korii.
183      * Koriin osuessa tarkistetaan jäljellä oleva voima.
184      * Kiekon suunta riippuu annetusta suuntavoimasta.
185      */
186     @Override
187     public void actionPerformed(ActionEvent e) {
188         if (!jatkuu && !liianlujaa) {
189             return;
190         }
191 
192         if (!jatkuu) {
193             try {
194                 Thread.sleep(2000);
195             } catch (Exception b) {
196             }
197 
198             this.liianlujaa = false;
199         }
200 
201         if (heitto.getTiiaus()) {
202 
203 //            if (heitto.getSuunta() != 0) {
204             if (heitto.getVoima() % heitto.getKerroin() == 0) {
205                 if (heitto.getSuunta() < 0) {
206                     kiekko.liiku(Suunta.VASEN);
207                     heitto.kasvataSuuntaa();
208                 } else if (heitto.getSuunta() > 0) {
209                     kiekko.liiku(Suunta.OIKEA);
210                     heitto.vahennaSuuntaa();
211                 }
212             }
213 //            }
214 
215             kiekko.liiku(Suunta.YLOS);
216             heitto.vahennaVoimaa();
217 
218             if (kori.osuu(kiekko.getSijainti())) {
219 
220                 if (heitto.getVoima() > 3) {
221                     pelaaja.lisaaHeitto();
222                     heitto.setVoima(1);
223                     this.liianlujaa = true;
224                 }
225 
226                 jatkuu = false;
227             }
228         }
229 
230         if (heitto.getVoima() == 0) {
231             heitto.setKerroin(0);
232             heitto.setTiiaus(false);
233         }
234 
235         paivitettava.paivita();
236         setDelay(300);
237 
238     }
