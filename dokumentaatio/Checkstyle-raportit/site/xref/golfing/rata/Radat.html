1   package golfing.rata;
2   
3   import golfing.kiekko.Kori;
4   import java.util.HashMap;
5   
6   /**
7    * Kiekkopelisovelluksen ratavalikoima.
8    * Tämä luokka on vastuussa eri ratojen säilytyksestä.
9    * Radat on kovakoodattu luokkaan.
10   *
11   */
12  
13  public class Radat {
14  
15      private HashMap<String, Rata> radat;
16      private int leveys;
17      private int korkeus;
18  
19      /**
20       * Radat valikoiman konstruktori.
21       * @param leveys Kiekkopelin leveys
22       * @param korkeus Kiekkopelin korkeus
23       * Tässä luodaan yksittäisiä ratoja, jotka sitten kuuluvat ratavalikoimaan.
24       * Luotuna 6 väyläinen Kumpula.
25       */
26      public Radat(int leveys, int korkeus) {
27          if (leveys <= 0 || korkeus <= 0) {
28              throw new IllegalArgumentException("Sekä leveyden, että korkeuden on oltava suurempi kuin nolla!");
29          }
30          this.radat = new HashMap<>();
31  
32          Rata kumpula = new Rata("Kumpulan frisbeegolfrata");
33          kumpula.lisaaVayla(new Vayla(new Kori(leveys / 2, 4), 2));
34          kumpula.lisaaVayla(new Vayla(new Kori(leveys / 2, korkeus / 2), 1));
35          kumpula.lisaaVayla(new Vayla(new Kori(leveys / 3, 1), 3));
36          kumpula.lisaaVayla(new Vayla(new Kori((leveys / 4) * 3, 2), 3));
37          kumpula.lisaaVayla(new Vayla(new Kori((leveys / 8) * 1, 2), 3));
38          kumpula.lisaaVayla(new Vayla(new Kori((leveys / 4) * 3, korkeus / 2), 2));
39          radat.put("Kumpula", kumpula);
40      }
41  
42      /**
43       * Ratavalikoima.
44       * @return Palauttaa ratavalikoiman.
45       */
46      public HashMap<String, Rata> getRadat() {
47          return radat;
48      }
49  }
