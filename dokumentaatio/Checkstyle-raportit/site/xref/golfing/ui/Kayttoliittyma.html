17  /**
18   * Kiekkopelisovelluksen graafinen käyttöliittymä.
19   *
20   */
21  public class Kayttoliittyma implements Runnable {
22  
23      private JFrame frame;
24      private Kiekkopeli kike;
25      private int pituus;
26      private Piirtoalusta piirtoalusta;
27  
28      /**
29       * Käyttöliittymä.
30       *
31       * @param kike Kiekkopeli-olio
32       * @param pituus pituus
33       */
34      public Kayttoliittyma(Kiekkopeli kike, int pituus) {
35          this.kike = kike;
36          this.pituus = pituus;
37  
38      }
39  
40      @Override
41      public void run() {
42          frame = new JFrame("Frisbeegolf .. MANIA!!");
43          int leveys = (kike.getLeveys() + 1) * pituus + 10;
44          int korkeus = (kike.getKorkeus() + 1) * pituus + 10;
45  
46          frame.setPreferredSize(new Dimension(leveys, korkeus));
47  
48          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
49  
50          luoKomponentit(frame.getContentPane());
51  
52          frame.pack();
53          frame.setVisible(true);
54  
55      }
56  
57      private void luoKomponentit(Container container) {
58  
59          CardLayout layout = new CardLayout();
60          JPanel kortit = new JPanel(layout);
61  
62          JButton alku = new JButton("Aloitusnäyttö");
63          JButton menu = new JButton("Menu");
64          JButton peli = new JButton("Peli");
65          JButton tuloskortti = new JButton("Tuloskortti");
66  
67          JPanel nappiPaneeli = new JPanel();
68  
69          lisaaNapitPaneeliin(nappiPaneeli, alku, menu, peli, tuloskortti);
70  
71          lisaaKuuntelijatValilehtiNapeille(alku, layout, kortit, menu, peli, tuloskortti);
72  
73          Nappaimistonkuuntelija kuuntelija = new Nappaimistonkuuntelija(kike.getPelaaja());
74          peli.addKeyListener(kuuntelija);
75  
76          BorderLayout alkulayout = new BorderLayout();
77          JTextField pelaajanNimi = new JTextField("Syötä tähän nimesi");
78          JTextField radanNimi = new JTextField("Syötä tähän valitsemasi rata, vaihtoehdot tänään: "
79                  + kike.getRadat().getRadat().keySet());
80          JButton vahvistusNappi = new JButton("Vahvista tästä, ja siirry peli-ikkunaan hiirellä");
81  
82  //        JButton ohje = new JButton("Ohjeet");
83  //        JButton kiekonValinta = new JButton("Valitse kiekot");
84          vahvistusNappi.addActionListener(new Alkunaytonkuuntelija(pelaajanNimi, radanNimi, kike, layout, kortit));
85  
86          JPanel alkupaneeli = new JPanel(alkulayout);
87          alkupaneeli.add(pelaajanNimi, BorderLayout.NORTH);
88          alkupaneeli.add(radanNimi, BorderLayout.SOUTH);
89          alkupaneeli.add(vahvistusNappi, BorderLayout.CENTER);
90  //        alkupaneeli.add(ohje, BorderLayout.WEST);
91  //        alkupaneeli.add(kiekonValinta, BorderLayout.EAST);
92          kortit.add(alkupaneeli, "alkuvalikko");
93  
94          piirtoalusta = new Piirtoalusta(kike, pituus);
95          kortit.add(piirtoalusta, "pelinakyma");
96  
97          BorderLayout menuLayout = new BorderLayout();
98          JPanel menupaneeli = new JPanel(menuLayout);
99  
100         ImageIcon kumpulakartta = new ImageIcon("src/main/resources/kumpulakartta.jpeg");
101         JButton vaylanVaihtoNappi = new JButton(kumpulakartta);
102         vahvistusNappi.setName("Vaihda väylää");
103         vaylanVaihtoNappi.addActionListener(new Vaylanapinkuuntelija(kike));
104 
105         JButton kiekonVaihtoNappi = new JButton("Vaihda kiekkoa");
106         kiekonVaihtoNappi.addActionListener(new Kiekonvaihtonapinkuuntelija(kike, kiekonVaihtoNappi));
107 
108         menupaneeli.add(vaylanVaihtoNappi, BorderLayout.CENTER);
109         menupaneeli.add(kiekonVaihtoNappi, BorderLayout.NORTH);
110 
111         kortit.add(menupaneeli, "menu");
112 
113         Tuloskortinpiirtaja tuloskortinpiirtaja = new Tuloskortinpiirtaja(kike);
114         kortit.add(tuloskortinpiirtaja, "tuloskortti");
115         tuloskortti.addActionListener(new Tuloskortinkuuntelija(tuloskortinpiirtaja));
116 
117         kike.setPaivitettava(piirtoalusta);
118 
119         frame.add(kortit);
120 
121         frame.add(nappiPaneeli, BorderLayout.SOUTH);
122 
123 //        container.add(piirtoalusta);
124 //        frame.addKeyListener(kuuntelija);
125     }
126 
127     private void lisaaNapitPaneeliin(JPanel nappiPaneeli, JButton alku, JButton menu, JButton peli, JButton tuloskortti) {
128         nappiPaneeli.add(alku);
129         nappiPaneeli.add(menu);
130         nappiPaneeli.add(peli);
131         nappiPaneeli.add(tuloskortti);
132     }
133 
134     private void lisaaKuuntelijatValilehtiNapeille(JButton alku, CardLayout layout, JPanel kortit, JButton menu, JButton peli, JButton tuloskortti) {
135         alku.addActionListener(new Napinkuuntelija(layout, kortit));
136         menu.addActionListener(new Napinkuuntelija(layout, kortit));
137         peli.addActionListener(new Napinkuuntelija(layout, kortit));
138         tuloskortti.addActionListener(new Napinkuuntelija(layout, kortit));
139     }
140

142      *
143      * @return palauttaa piirtoalustan
144      */
145     public Paivitettava getPaivitettava() {
146         return piirtoalusta;
147     }
148 
149     /**
150      *
151      * @return framen
152      */
153     public JFrame getFrame() {
154         return frame;
155     }
