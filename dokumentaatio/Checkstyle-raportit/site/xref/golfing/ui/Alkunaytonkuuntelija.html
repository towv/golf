1   package golfing.ui;
2   
3   import golfing.kiekko.Pelaaja;
4   import golfing.peli.Kiekkopeli;
5   import java.awt.CardLayout;
6   import java.awt.event.ActionEvent;
7   import java.awt.event.ActionListener;
8   import javax.swing.JPanel;
9   import javax.swing.JTextField;
10  
11  /**
12   *
13   * Alkunäytönkuuntelija. Kuuntelee alkunäyttöä. 
14   * Käyttäjän nimi ja radannimi käyttäjän syötteen mukaan.
15   */
16  public class Alkunaytonkuuntelija implements ActionListener {
17  
18      private JTextField kayttajanNimi;
19      private JTextField radanNimi;
20      private Kiekkopeli kiekkopeli;
21      private CardLayout layout;
22      private JPanel kortit;
23  
24      /**
25       * Konstruktori. 
26       * @param kayttajanNimi Tekstialue käyttäjän nimen syöttämistä varten.
27       * @param radanNimi Tekstialue radan nimen syöttämistä varten.
28       * @param kiekkopeli Pelattava peli
29       * @param layout Kortti layout.
30       * @param kortit Korttien piirto.
31       */
32      public Alkunaytonkuuntelija(JTextField kayttajanNimi, JTextField radanNimi, Kiekkopeli kiekkopeli, CardLayout layout, JPanel kortit) {
33          this.kayttajanNimi = kayttajanNimi;
34          this.radanNimi = radanNimi;
35          this.kiekkopeli = kiekkopeli;
36          this.layout = layout;
37          this.kortit = kortit;
38      }
39  
40      @Override
41      public void actionPerformed(ActionEvent e) {
42          boolean kayttajanNimiSyotetty = false;
43  
44          if (kayttajanNimi.getText().isEmpty()) {
45              kayttajanNimi.setText("Nimeä ei voi jättää tyhjäksi!");
46          } else if (kayttajanNimi.getText().equals("Nimeä ei voi jättää tyhjäksi!") 
47                  || kayttajanNimi.getText().equals("Syötä tähän nimesi")
48                  || kayttajanNimi.getText().equals("Anna nimesi")) {
49              kayttajanNimi.setText("Anna nimesi");
50          } else {
51              kiekkopeli.getPelaaja().setNimi(kayttajanNimi.getText());
52              kayttajanNimiSyotetty = true;
53          }
54  
55          String rata = radanNimi.getText();
56          boolean valittuOnRata = false;
57          for (String s : kiekkopeli.getRadat().getRadat().keySet()) {
58              if (rata.equals(s)) {
59                  valittuOnRata = true;
60              }
61          }
62          if (valittuOnRata) {
63              kiekkopeli.setRata(rata);
64          } else if (rata.equals("Valitsemasi nimistä rataa ei ole tänään tarjolla!")) {
65              radanNimi.setText("Ratavaihtoehdot tänään " + kiekkopeli.getRadat().getRadat().keySet());
66          } else {
67              radanNimi.setText("Valitsemasi nimistä rataa ei ole tänään tarjolla!");
68          }
69  
70          if (kayttajanNimiSyotetty && valittuOnRata) {
71              layout.show(kortit, "pelinakyma");
72          }
73  
74      }
75  
76  }
