1   package golfing.ui;

3   import golfing.peli.Kiekkopeli;
4   import golfing.peli.Tuloskortti;
5   import java.awt.Color;
6   import java.awt.Font;
7   import java.awt.Graphics;
8   import java.util.ArrayList;
9   import java.util.HashMap;
10  import javax.swing.JPanel;
11  
12  /**
13   *
14   * Tuloskortin piirtoalusta. Piirtää tulokset kiekkopelin mukaan.
15   */
16  public class Tuloskortinpiirtaja extends JPanel implements Paivitettava {
17  
18      private Kiekkopeli kike;
19  
20      /**
21       * Tuloskortin konstruktori.
22       *
23       * @param kike Kiekkopeli-olio.
24       */
25      public Tuloskortinpiirtaja(Kiekkopeli kike) {
26          super.setBackground(Color.CYAN);
27          this.kike = kike;
28      }
29  
30      @Override
31      protected void paintComponent(Graphics g) {
32          super.paintComponent(g);
33  
34          Tuloskortti tuloskortti = kike.getTuloskortti();
35  
36          g.setColor(Color.red);
37          g.setFont(new Font("Comic Sans MS", Font.BOLD, 26));
38  
39          g.drawString("" + tuloskortti.getTulosRata(), 10, 25);
40  
41          g.setColor(Color.yellow);
42          g.setFont(new Font("Comic Sans MS", Font.BOLD, 20));
43  
44          HashMap<Integer, String> tulokset = tuloskortti.getTuloksetString();
45          int i = 1;
46          for (String value : tulokset.values()) {
47              i++;
48              g.drawString(value, 5, 40 + 20 * i);
49          }
50  
51  //        for (int i = 1; i <= tulokset.size(); i++) {
52  //            g.drawString(tulokset.get(i), 5, 40 + 20 * i);
53  //        }
54          g.setColor(Color.blue);
55          g.setFont(new Font("Comic Sans MS", Font.BOLD, 20));
56          g.drawString("Ihanne heitot: " + tuloskortti.ihanneTulos()
57                  + "  Pelaajan: " + tuloskortti.yhteisTulos(), 0, 350);
58  
59          int tulos = tuloskortti.yhteisTulos() - tuloskortti.ihanneTulos();
60          String plus = "";
61          if (tulos > 0) {
62              plus = "+";
63          }
64          g.setColor(Color.blue);
65          g.setFont(new Font("Comic Sans MS", Font.BOLD, 20));
66          g.drawString("Pelaajan tulos: " + plus + tulos, 0, 380);
67  
68          if (tuloskortti.radanPituus() == tulokset.size()) {
69              g.setColor(Color.magenta);
70              g.setFont(new Font("Comic Sans MS", Font.BOLD, 30));
71              g.drawString("KIERROS PELATTU", 0, 500);
72          }
73      }
74  
75      /**
76       * Paivitettava rajapinnan paivita() metodi ylikirjoitetaan.
77       */
78      @Override
79      public void paivita() {
80          repaint();
81      }
82  }
