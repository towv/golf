1   package golfing.kiekko;
2   
3   import java.util.HashMap;
4   
5   /**
6    * Pelaaja luokka kuvastaa käyttäjää. Käyttäjällä on nimi, bägillinen kiekkoja, 
7    * heitto ja sijainti kartalla. Pelaaja luokka pitää myös kirjaa käyttäjän 
8    * suorittamien heittojen määrästä. Pelaajalla voi olla useita kiekkoja joten se pitää 
9    * tietoa käytössä olevan kiekon väristä muistissa.
10   *
11   */
12  
13  public class Pelaaja {
14  
15      private String nimi;
16      private HashMap<String, Kiekko> kiekot;
17      private Sijainti sijainti;
18      private int heitot;
19      private Heitto heitto;
20      private Vari kaytossaOlevanKiekonVari;
21  
22      public Pelaaja(String nimi, int leveys, int korkeus) {
23          this.nimi = nimi;
24          this.kiekot = new HashMap<>();
25          this.sijainti = new Sijainti(leveys / 2, korkeus - 1);
26          this.heitot = 0;
27          this.heitto = new Heitto(0, 0);
28      }
29  
30      public void lisaaKiekko() {
31          kiekot.put("draiveri", new Kiekko("draiveri", sijainti.getX(), sijainti.getY()));
32      }
33  
34      public Kiekko getKiekko(String nimi) {
35          return kiekot.get(nimi);
36      }
37  
38      public void lisaaHeitto() {
39          heitot++;
40      }
41  
42      public int montakoHeittoa() {
43          return heitot;
44      }
45  
46      public String getNimi() {
47          return nimi;
48      }
49  
50      public Heitto getHeitto() {
51          return heitto;
52      }
53  
54      public Vari getKaytossaOlevanKiekonVari() {
55          kaytossaOlevanKiekonVari = getKiekko("draiveri").getVari();
56          return kaytossaOlevanKiekonVari;
57      }
58  
59      public Sijainti getSijainti() {
60          return sijainti;
61      }
62  }
