1   package golfing.gui;
2   
3   import golfing.kiekko.Vari;
4   import golfing.peli.Kiekkopeli;
5   import java.awt.Color;
6   import java.awt.Font;
7   import java.awt.Graphics;
8   import javax.swing.JPanel;
9   
10  /**
11   * Kiekkopelisovelluksen piirtäjä. Toteuttaa paivitettava rajapinnan.
12   *
13   */
14  
15  public class Piirtoalusta extends JPanel implements Paivitettava {
16  
17      private Kiekkopeli kiekkopeli;
18      private int kiekonLeveys;
19      private Vari kiekonVari;
20  
21      public Piirtoalusta(Kiekkopeli kiekkopeli, int kiekonLeveys) {
22          super.setBackground(Color.CYAN);
23          this.kiekkopeli = kiekkopeli;
24          this.kiekonLeveys = kiekonLeveys;
25          this.kiekonVari = kiekkopeli.getPelaaja().getKaytossaOlevanKiekonVari();
26      }
27  
28      @Override
29      protected void paintComponent(Graphics g) {
30          super.paintComponent(g);
31  
32          g.setColor(Color.red);
33          g.setFont(new Font("Comic Sans MS", Font.PLAIN, 20));
34          g.drawString("" + kiekkopeli.getTilanne(), 0, 20);
35          
36          String viesti = kiekkopeli.getViesti();
37          String[] rivit = viesti.split("\n");
38          String rivi1 = rivit[0];
39          String rivi2 = rivit[1];
40          
41          g.setFont(new Font("Comic Sans MS", Font.BOLD, 24));
42          g.drawString(rivi1, 0, 350);
43          g.setFont(new Font("Comic Sans MS", Font.PLAIN, 20));
44          g.drawString(rivi2, 0, 380);
45  
46          g.setColor(Color.ORANGE);
47          g.fillRect(130, 210, 15, 100);
48  
49          g.setColor(Color.GREEN);
50          g.fillOval(100, 200, 80, 40);
51  
52          g.setColor(Color.BLACK);
53          g.fillOval(kiekkopeli.getKori().getX() * kiekonLeveys, kiekkopeli.getKori().getY() * kiekonLeveys, kiekonLeveys * 2, kiekonLeveys * 2);
54          
55          if (kiekonVari == Vari.LIILA) {
56              g.setColor(Color.MAGENTA);
57          } else if (kiekonVari == Vari.HARMAA) {
58              g.setColor(Color.GRAY);
59          } else if (kiekonVari == Vari.SININEN) {
60              g.setColor(Color.BLUE);
61          }
62          
63          g.fillOval(kiekkopeli.getKiekko().getSijainti().getX() * kiekonLeveys, kiekkopeli.getKiekko().getSijainti().getY() * kiekonLeveys, kiekonLeveys, kiekonLeveys);
64      }
65  
66      @Override
67      public void paivita() {
68          repaint();
69      }
70  
71  }
